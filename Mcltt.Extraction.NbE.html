Inductive initial_env_order : ctx -> Prop :=
| ie_nil : initial_env_order nil
| ie_cons :
  `( initial_env_order Γ ->
     (forall p, initial_env Γ p ->
           eval_exp_order A p) ->
     initial_env_order (A :: Γ)).

Inductive nbe_order G M A : Prop :=

  `( initial_env_order G ->
     (forall p, initial_env G p ->
           eval_exp_order A p) ->
     (forall p, initial_env G p ->
           eval_exp_order M p) ->
     (forall p a m,
         initial_env G p ->
         {{ ⟦ A ⟧ p ↘ a }} ->
